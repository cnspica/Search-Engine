        function subscribe(email) {

                url: '/newsletters/subscribe',

                dataType: 'json',
                success: function (data) {
                    $('.promotion-banner').html('<div class="promotion-message"><p><strong>You have subscribed to the Game Development Weekly Newsletter!</strong></p></div>');

                    StackExchange.helpers.removeSpinner();
                }
            });
