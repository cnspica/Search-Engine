                data: { 'fkey': '8c860d19b546a5f1bec5ccc0119119dd', 'siteId': 259, 'email': savedEmail },
                success: function() {
                    $that.replaceWith('<strong>sent!</strong>');
                },
                complete: function () {
                    StackExchange.helpers.removeSpinner();
                }
            });

            return false;
        });

        function subscribe(email) {
            StackExchange.helpers.addSpinner('.subscribe-button-container');

            $.ajax({
                type: 'POST',
                url: '/newsletters/subscribe',
                data: { 'fkey': '8c860d19b546a5f1bec5ccc0119119dd', 'siteId': 259, 'email': email },
                dataType: 'json',
                success: function (data) {
                    $('.promotion-banner').html('<div class="promotion-message"><p><strong>You have subscribed to the Sound Design Weekly Newsletter!</strong></p></div>');
                    
                    if (data == "unconfirmed") {

                },
                complete: function () {
                    StackExchange.helpers.removeSpinner();
                }
            });
        }
