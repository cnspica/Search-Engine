        function subscribe(email) {

                type: 'POST',
                url: '/newsletters/subscribe',

                success: function (data) {
                    $('.promotion-banner').html('<div class="promotion-message"><p><strong>You have subscribed to the Information Security Weekly Newsletter!</strong></p></div>');
