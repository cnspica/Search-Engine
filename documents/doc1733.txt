   <script>
   $(document).ready(function(){
      var regEmail =  /^([\w-\.]+)@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.)|(([\w-]+\.)+))([a-zA-Z]{2,4}|[0-9]{1,3})(\]?)$/;
      $('#submitForgot').on('click', function() {   
         var email = $('#textEmail').val();        
         if (email == '')
         { 
            $('#textEmail').next().html('Please enter a email id');
            $('#textEmail').focus();
            $('#textEmail').css("border-color","red");
            return false;
         } 
         else if (!regEmail.test(email))
         { 
            $('#textEmail').next().html('Please enter a valid email id');
            $('#textEmail').focus();
             $('#textEmail').css("border-color","red");
            return false;
         }
        
         $('#textEmail').next().html('');
         $('#textEmail').css("border-color","#ccc");
         
         $.ajax({
            url: baseurl+"ajax/ajaxUserForgot.php",
            method:"POST",
            data: {"email":email}
         }).done(function(msg) { 
            if(msg == -1)
            {    
               $('#errDiv').html('Email id you entered is not registered with us!');	
            }
            if(msg == 0)
            {
               $('#errDiv').html('Some thing went wrong. Please try after some time.');
               return false;
            }
            if(msg == 1)
            {	
               $('#errDiv').css('color','#024622');  
               $('#errDiv').html('New password is sent to your email id.');  
               $('#textEmail').val('');
               return false;
            }             
         });      
      });
   });
   </script>
