                if (q.ownerDisplayName) {
                    if (q.ownerUrl) {
                        $question.find('.realtime-owner').html('<a href="' + q.ownerUrl + '">' + q.ownerDisplayName + '</a> on');
                    }
                    else {
                        $question.find('.realtime-owner').text(q.ownerDisplayName + ' on');
                    }
                }
                else {
                    $question.find('.realtime-owner').remove();
                }

                $question.find('.realtime-host').attr('href', 'http://' + q.siteBaseHostAddress).text(sitesData[q.siteBaseHostAddress].hostShortName);
                
                $existingQuestion = $('.' + realtimeKey);

                if ($existingQuestion.length) {
                    $existingQuestion.remove();
                }
                else if ($('.realtime-question:visible').length >= 30) {
                    $('.realtime-question:visible:last').remove();
                }

                updateRelativeTimes();

                $('#mainArea br.cbo').after($question.hide());
                $question.fadeIn();
            }

            function updateRelativeTimes () {
                $('.realtime-unix-time').each(function () {
                    var $this = $(this);
                    var relativeTime = StackExchange.realtime.getRelativeTime(parseInt($this.val(), 10));
                    $this.siblings('.realtime-activity-date').text(relativeTime);
                });
            }

            function getPollingUrl () {
                var $timeElem = $('.realtime-question:visible:first .realtime-unix-time');
                var time = $timeElem.length ? parseInt($timeElem.val(), 10) : Math.floor((new Date().getTime()) / 1000);
                return '/questions/poll-realtime?since=' + time;
            }
        });
        
    </script>
