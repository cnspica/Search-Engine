                data: { 'fkey': '8c860d19b546a5f1bec5ccc0119119dd', 'siteId': 249, 'email': savedEmail },
                success: function() {
                    $that.replaceWith('<strong>sent!</strong>');
                },
                complete: function () {
                    StackExchange.helpers.removeSpinner();
                }
            });

            return false;
        });

        function subscribe(email) {
            StackExchange.helpers.addSpinner('.subscribe-button-container');

            $.ajax({
                type: 'POST',
                url: '/newsletters/subscribe',
                data: { 'fkey': '8c860d19b546a5f1bec5ccc0119119dd', 'siteId': 249, 'email': email },
                dataType: 'json',
                success: function (data) {
                    $('.promotion-banner').html('<div class="promotion-message"><p><strong>You have subscribed to the Blender Weekly Newsletter!</strong></p></div>');
                    
                    if (data == "unconfirmed") {
                        savedEmail = email;
                        $('.promotion-message').append('<p>Please click the link in the confirmation email to activate your subscription.</p><p><a href="#" id="resend-confirmation">(re-send confirmation)</a></p>');

                },
                error: function (jqXHR, textStatus, errorThrown) {

                },
                complete: function () {
                    StackExchange.helpers.removeSpinner();
                }
            });
        }
    });
