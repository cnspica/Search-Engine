 $(function() {
       if (localStorage.chkbx && localStorage.chkbx != '') {
           $('#remember_me').attr('checked', 'checked');
           $('#textEmail').val(localStorage.useremail);
           $('#textPassword').val(localStorage.pwd);
       } else {
           $('#remember_me').removeAttr('checked');
           $('#textEmail').val('');
           $('#textPassword').val('');
       }

       $('#remember_me').click(function() {

           if ($('#remember_me').is(':checked')) {               
               localStorage.useremail = $('#textEmail').val();
               localStorage.pwd = $('#textPassword').val();
               localStorage.chkbx = $('#remember_me').val();
           } else {
               localStorage.useremail = '';
               localStorage.pwd = '';
               localStorage.chkbx = '';
           }
       });
   });
