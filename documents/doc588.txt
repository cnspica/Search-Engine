        function subscribe(email) {

                success: function (data) {
                    $('.promotion-banner').html('<div class="promotion-message"><p><strong>You have subscribed to the Raspberry Pi Weekly Newsletter!</strong></p></div>');

                    StackExchange.helpers.removeSpinner();
                }
            });
